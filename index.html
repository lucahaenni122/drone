#include #include #include #include // ── WLAN Zugangsdaten ───────────────────────────────────────────────────── const char* SSID = "FRITZ!Box 4040 EE"; // ← anpassen const char* PASS = "89243552272146594014"; // ← anpassen // ── Motor Pins ──────────────────────────────────────────────────────────── const int MOTOR1 = 12; // Vorne Links const int MOTOR2 = 13; // Vorne Rechts const int MOTOR3 = 14; // Hinten Links const int MOTOR4 = 15; // Hinten Rechts // ── Geschwindigkeiten ───────────────────────────────────────────────────── #define BASE_SPEED 180 #define HOVER_SPEED 140 #define TURN_DIFF 40 // ── Timeout ─────────────────────────────────────────────────────────────── #define CMD_TIMEOUT_MS 300 // ── State ───────────────────────────────────────────────────────────────── bool motorsEnabled = false; unsigned long lastCmdTime = 0; AsyncWebServer server(80); AsyncWebSocket ws("/ws"); // ── Hilfsfunktionen ─────────────────────────────────────────────────────── void setMotor(int pin, int speed) { analogWrite(pin, constrain(speed, 0, 255)); } void allMotorsOff() { setMotor(MOTOR1, 0); setMotor(MOTOR2, 0); setMotor(MOTOR3, 0); setMotor(MOTOR4, 0); } // ── Bewegungsfunktionen ─────────────────────────────────────────────────── void cmdForward() { setMotor(MOTOR1, BASE_SPEED + TURN_DIFF); setMotor(MOTOR2, BASE_SPEED + TURN_DIFF); setMotor(MOTOR3, BASE_SPEED - TURN_DIFF); setMotor(MOTOR4, BASE_SPEED - TURN_DIFF); } void cmdBack() { setMotor(MOTOR1, BASE_SPEED - TURN_DIFF); setMotor(MOTOR2, BASE_SPEED - TURN_DIFF); setMotor(MOTOR3, BASE_SPEED + TURN_DIFF); setMotor(MOTOR4, BASE_SPEED + TURN_DIFF); } void cmdLeft() { setMotor(MOTOR1, BASE_SPEED - TURN_DIFF); setMotor(MOTOR2, BASE_SPEED + TURN_DIFF); setMotor(MOTOR3, BASE_SPEED - TURN_DIFF); setMotor(MOTOR4, BASE_SPEED + TURN_DIFF); } void cmdRight() { setMotor(MOTOR1, BASE_SPEED + TURN_DIFF); setMotor(MOTOR2, BASE_SPEED - TURN_DIFF); setMotor(MOTOR3, BASE_SPEED + TURN_DIFF); setMotor(MOTOR4, BASE_SPEED - TURN_DIFF); } void cmdUp() { int spd = constrain(BASE_SPEED + 30, 0, 255); setMotor(MOTOR1, spd); setMotor(MOTOR2, spd); setMotor(MOTOR3, spd); setMotor(MOTOR4, spd); } void cmdDown() { int spd = constrain(BASE_SPEED - 30, 0, 255); setMotor(MOTOR1, spd); setMotor(MOTOR2, spd); setMotor(MOTOR3, spd); setMotor(MOTOR4, spd); } void cmdRotLeft() { setMotor(MOTOR1, BASE_SPEED + TURN_DIFF); setMotor(MOTOR2, BASE_SPEED - TURN_DIFF); setMotor(MOTOR3, BASE_SPEED - TURN_DIFF); setMotor(MOTOR4, BASE_SPEED + TURN_DIFF); } void cmdRotRight() { setMotor(MOTOR1, BASE_SPEED - TURN_DIFF); setMotor(MOTOR2, BASE_SPEED + TURN_DIFF); setMotor(MOTOR3, BASE_SPEED + TURN_DIFF); setMotor(MOTOR4, BASE_SPEED - TURN_DIFF); } // ── Befehl verarbeiten ──────────────────────────────────────────────────── void processCommand(String cmd) { cmd.trim(); lastCmdTime = millis(); if (cmd == "MOTORS_ON") { motorsEnabled = true; // Motoren bleiben aus – starten erst beim ersten Steuerbefehl ws.textAll("STATUS:MOTORS_ON"); ws.textAll("OK:Motoren bereit"); return; } if (cmd == "MOTORS_OFF" || cmd == "STOP") { if (cmd == "MOTORS_OFF") motorsEnabled = false; allMotorsOff(); ws.textAll("STATUS:MOTORS_OFF"); ws.textAll("OK:Gestoppt"); return; } if (!motorsEnabled) { ws.textAll("ERR:Motoren nicht aktiv"); return; } if (cmd == "FWD") { cmdForward(); ws.textAll("OK:Vorwaerts"); } else if (cmd == "BACK") { cmdBack(); ws.textAll("OK:Rueckwaerts"); } else if (cmd == "LEFT") { cmdLeft(); ws.textAll("OK:Links"); } else if (cmd == "RIGHT"){ cmdRight(); ws.textAll("OK:Rechts"); } else if (cmd == "UP") { cmdUp(); ws.textAll("OK:Steigen"); } else if (cmd == "DOWN") { cmdDown(); ws.textAll("OK:Sinken"); } else if (cmd == "ROTL") { cmdRotLeft(); ws.textAll("OK:Drehen_L"); } else if (cmd == "ROTR") { cmdRotRight(); ws.textAll("OK:Drehen_R"); } else { ws.textAll("ERR:Unbekannter_Befehl:" + cmd); } } // ── WebSocket Event Handler ─────────────────────────────────────────────── void onWsEvent(AsyncWebSocket* s, AsyncWebSocketClient* c, AwsEventType t, void* arg, uint8_t* data, size_t len) { if (t == WS_EVT_CONNECT) { Serial.printf("WS Client verbunden: %u\n", c->id()); ws.textAll("SYS:Drohnen-Controller bereit"); } else if (t == WS_EVT_DISCONNECT) { Serial.printf("WS Client getrennt: %u\n", c->id()); allMotorsOff(); motorsEnabled = false; } else if (t == WS_EVT_DATA) { String cmd = ""; for (size_t i = 0; i < len; i++) cmd += (char)data[i]; Serial.println("CMD: " + cmd); processCommand(cmd); } } // ── Setup ───────────────────────────────────────────────────────────────── void setup() { Serial.begin(115200); pinMode(MOTOR1, OUTPUT); pinMode(MOTOR2, OUTPUT); pinMode(MOTOR3, OUTPUT); pinMode(MOTOR4, OUTPUT); allMotorsOff(); // Fixe IP IPAddress ip(192, 168, 1, 100); IPAddress gateway(192, 168, 1, 1); IPAddress subnet(255, 255, 255, 0); WiFi.config(ip, gateway, subnet); // WLAN verbinden WiFi.begin(SSID, PASS); Serial.print("Verbinde mit WLAN"); while (WiFi.status() != WL_CONNECTED) { delay(500); Serial.print("."); } Serial.println("\nVerbunden! IP: " + WiFi.localIP().toString()); // WebSocket ws.onEvent(onWsEvent); server.addHandler(&ws); server.begin(); Serial.println("WebSocket Server gestartet auf ws://" + WiFi.localIP().toString() + "/ws"); } // ── Loop ────────────────────────────────────────────────────────────────── void loop() { ws.cleanupClients(); // Timeout: Motoren aus wenn kein Befehl if (motorsEnabled && (millis() - lastCmdTime) > CMD_TIMEOUT_MS) { allMotorsOff(); } }
